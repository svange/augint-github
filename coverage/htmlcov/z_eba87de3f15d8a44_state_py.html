110# Golden-angle hue step -- guarantees that every new slot lands maximally far 

111# from the previously-used hues, so adjacent teams can never collide into 

112# similar colours (e.g. "woxom" vs "augmenting-integrations" both being blue). 

113_GOLDEN_ANGLE = 137.50776405003785 

114# Fixed saturation / lightness -- tuned for terminal legibility against both 

115# dark and paper-like backgrounds. 

116_TEAM_ACCENT_S = 0.60 

117_TEAM_ACCENT_L = 0.62 

118

120def _hsl_hex(hue_deg: float) -> str: 

121 h = (hue_deg % 360.0) / 360.0 

122 r, g, b = colorsys.hls_to_rgb(h, _TEAM_ACCENT_L, _TEAM_ACCENT_S) 

123 return f"#{int(r * 255):02x}{int(g * 255):02x}{int(b * 255):02x}"

126def team_accent(team_key: str, known_teams: Iterable[str] | None = None) -> str: 

127 """Return a deterministic hex colour for ``team_key``. 

128 

129 When ``known_teams`` is supplied, colours are assigned by *sorted index* 

130 through the golden-angle rotation, so every team in that set is guaranteed 

131 a maximally-distinct hue from its neighbours. Without it we fall back to 

132 hashing the team key into the same rotation -- still deterministic per 

133 key, but no longer guaranteed distinct between a given pair of keys. 

134 """ 

135 if team_key == UNASSIGNED_TEAM: 

136 return "#808080" 

137 if known_teams is not None: 

138 keys = sorted({k for k in known_teams if k and k != UNASSIGNED_TEAM}) 

139 if team_key in keys: 

140 idx = keys.index(team_key) 

141 return _hsl_hex(idx * _GOLDEN_ANGLE) 

142 digest = hashlib.sha256(team_key.encode("utf-8")).digest() 

143 idx = int.from_bytes(digest[:2], "big") 

144 return _hsl_hex(idx * _GOLDEN_ANGLE)

153def remember_repo_teams(state: AppState, repo: Repository) -> None: 

154 """Populate state.repo_teams for a repo. Safe on API failure.""" 

155 full_name = getattr(repo, "full_name", "") 

156 if not full_name or full_name in state.repo_teams: 

157 return 

158 try: 

159 teams = sorted(repo.get_teams(), key=_team_sort_key) 

160 except Exception: 

161 state.repo_teams[full_name] = RepoTeamInfo() 

162 return 

163 

164 team_keys: list[str] = [] 

165 for team in teams: 

166 slug = getattr(team, "slug", "") or "" 

167 if not slug: 

168 continue 

169 name = getattr(team, "name", "") or slug 

170 team_keys.append(slug) 

171 state.team_labels.setdefault(slug, name) 

172 

173 if not team_keys: 

174 state.repo_teams[full_name] = RepoTeamInfo() 

175 return 

176 state.repo_teams[full_name] = RepoTeamInfo(primary=team_keys[0], all=tuple(team_keys))

192def apply_filter( 

193 healths: list[RepoHealth], 

194 mode: str, 

195 repo_teams: dict[str, RepoTeamInfo] | None = None, 

196) -> list[RepoHealth]: 

197 repo_teams = repo_teams or {} 

198 if mode == "all": 

199 return list(healths) 

200 if mode == "broken-ci": 

201 return [ 

202 h 

203 for h in healths 

204 if any(c.check_name == "broken_ci" and c.severity != Severity.OK for c in h.checks) 

205 ] 

206 if mode == "no-renovate": 

207 return [ 

208 h 

209 for h in healths 

210 if any( 

211 c.check_name == "renovate_enabled" and c.severity != Severity.OK for c in h.checks 

212 ) 

213 ] 

214 if mode == "stale-prs": 

215 return [ 

216 h 

217 for h in healths 

218 if any(c.check_name == "stale_prs" and c.severity != Severity.OK for c in h.checks) 

219 ] 

220 if mode == "issues": 

221 return [ 

222 h 

223 for h in healths 

224 if any(c.check_name == "open_issues" and c.severity != Severity.OK for c in h.checks) 

225 ] 

226 team_key = team_key_from_filter(mode) 

227 if team_key is not None: 

228 if team_key == UNASSIGNED_TEAM: 

229 return [ 

230 h 

231 for h in healths 

232 if repo_teams.get(h.status.full_name, RepoTeamInfo()).primary == UNASSIGNED_TEAM 

233 ] 

234 return [ 

235 h for h in healths if team_key in repo_teams.get(h.status.full_name, RepoTeamInfo()).all 

236 ] 

237 return list(healths)

240def available_filter_modes( 

241 team_labels: dict[str, str], repo_teams: dict[str, RepoTeamInfo] 

242) -> list[str]: 

243 team_keys = {team for info in repo_teams.values() for team in (info.all or (info.primary,))} 

244 dynamic = [ 

245 team_filter_mode(team_key) 

246 for team_key in sorted( 

247 team_keys, key=lambda key: display_team_label(key, team_labels).lower() 

248 ) 

249 ] 

250 return [*FILTER_MODES, *dynamic]

276def move_selection(state: AppState, delta: int) -> None: 

277 vis = visible_healths(state) 

278 if not vis: 

279 return 

280 ensure_selection(state) 

281 current_index = next( 

282 (i for i, h in enumerate(vis) if h.status.full_name == state.selected_full_name), 

283 0, 

284 ) 

285 new_index = max(0, min(len(vis) - 1, current_index + delta)) 

286 state.selected_full_name = vis[new_index].status.full_name

294def bootstrap_from_cache(state: AppState, restrict_to: set[str] | None = None) -> bool: 

295 """Populate state.healths from the on-disk cache. Returns True if loaded.""" 

296 try: 

297 cache = load_cache() 

298 if restrict_to is not None: 

299 cache = {k: v for k, v in cache.items() if k in restrict_to} 

300 if not cache: 

301 return False 

302 statuses: list[RepoStatus] = list(cache.values()) 

303 health_cache = load_health_cache(cache) 

304 state.healths = [health_cache.get(s.full_name) or RepoHealth(status=s) for s in statuses] 

305 state.health_by_name = {h.status.full_name: h for h in state.healths} 

306 return True 

307 except Exception as exc: 

308 state.log_error("cache", f"{exc.__class__.__name__}: {exc}") 

309 return False
